## 144. Binary Tree Preorder Traversal

## c++
<pre>
  class Solution {
public:
    vector<int> preorderTraversal(TreeNode* root) {
        vector<int> ans;
        stack<TreeNode*> st;

        while (root || !st.empty()) {
            if (root == 0) {
               root = st.top();
               st.pop();
            }
            ans.push_back(root->val);

            if (root->right) st.push(root->right);
            root = root->left;
        }
        return ans;
    }
};
