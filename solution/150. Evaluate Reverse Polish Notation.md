# 150. Evaluate Reverse Polish Notation

## C++
<pre>
  class Solution {
public:
    int evalRPN(vector<string>& tokens) {
        stack<int> st;

        for (string i : tokens) {
            if (i != "+" && i != "-" && i != "*" && i != "/") {
                printf(" %d\n", stoi(i));
                st.push(stoi(i));
            } else {
                int a, b;
                b = st.top();
                st.pop();
                a = st.top();
                st.pop();
        printf("%d %d\n", a, b);

                if (i == "+") st.push(a + b);
                else if (i == "-") st.push(a - b);
                else if (i == "*") st.push(a * b);
                else if (i == "/") st.push(a / b);
            }
        }
        return st.top();
    }
};
